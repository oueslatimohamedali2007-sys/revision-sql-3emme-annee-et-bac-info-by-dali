Chapitre 1

Définition des données (DDL)

Commandes pour structurer, créer et modifier les bases de données et les tables.

CREATE DATABASE — Créer une Base de Données

Crée une nouvelle base de données vide.

-- Créer une base de données simple
CREATE DATABASE ecole_db;

-- Créer si n'existe pas déjà
CREATE DATABASE IF NOT EXISTS ecole_db;

-- Créer avec encodage spécifique
CREATE DATABASE ecole_db CHARACTER SET utf8mb4;

DROP DATABASE — Supprimer une Base de Données

Supprime complètement une base de données et tous ses contenus.

-- Supprimer une base de données
DROP DATABASE ecole_db;

-- Supprimer si existe
DROP DATABASE IF EXISTS ecole_db;

-- ⚠️ Attention: Cette action est irréversible!

CREATE TABLE — Créer une Table

Génère une table avec ses colonnes et types associés.

-- Créer une table simple
CREATE TABLE etudiants (
  id INT PRIMARY KEY,
  nom VARCHAR(50) NOT NULL,
  prenom VARCHAR(50),
  classe VARCHAR(10),
  note_bac DECIMAL(4,2)
);

-- Avec auto-incrémentation
CREATE TABLE eleves (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nom VARCHAR(50) NOT NULL,
  email VARCHAR(100) UNIQUE,
  date_inscription DATE DEFAULT CURDATE()
);

ALTER TABLE — ADD — Ajouter une Colonne

Ajoute une nouvelle colonne à une table existante.

-- Ajouter une simple colonne
ALTER TABLE etudiants ADD email VARCHAR(100);

-- Ajouter avec contrainte
ALTER TABLE etudiants ADD telephone VARCHAR(20) UNIQUE;

-- Ajouter à une position précise
ALTER TABLE etudiants ADD age INT AFTER nom;

-- Ajouter au début
ALTER TABLE etudiants ADD id_classe INT FIRST;

ALTER TABLE — DROP — Supprimer une Colonne

Retire une colonne d'une table existante.

-- Supprimer une colonne
ALTER TABLE etudiants DROP COLUMN email;

-- Supprimer plusieurs colonnes (une à la fois)
ALTER TABLE etudiants DROP COLUMN telephone;
ALTER TABLE etudiants DROP COLUMN age;

ALTER TABLE — MODIFY — Modifier le Type d'une Colonne

Change le type de données d'une colonne existante.

-- Augmenter la taille d'une VARCHAR
ALTER TABLE etudiants MODIFY nom VARCHAR(100);

-- Changer le type de données
ALTER TABLE etudiants MODIFY note_bac DECIMAL(5,2);

-- Modifier avec NOT NULL
ALTER TABLE etudiants MODIFY prenom VARCHAR(50) NOT NULL;

ALTER TABLE — RENAME — Renommer

Renomme une table ou une colonne.

-- Renommer une colonne
ALTER TABLE etudiants RENAME COLUMN note_bac TO note_finale;

-- Renommer une table entière
ALTER TABLE etudiants RENAME TO eleves;

-- Renommer avec syntaxe alternative
RENAME TABLE etudiants TO eleves;

DROP TABLE — Supprimer une Table

Supprime complètement une table et ses données.

-- Supprimer une table
DROP TABLE etudiants;

-- Supprimer si existe
DROP TABLE IF EXISTS etudiants;

-- ⚠️ Attention: Toutes les données sont perdues!

💡 Sequence DDL Complète

Exemple complet de création d'une structure:

-- 1. Créer la base
CREATE DATABASE IF NOT EXISTS ma_school;
USE ma_school;

-- 2. Créer une table
CREATE TABLE classes (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nom VARCHAR(20) NOT NULL UNIQUE,
  niveau VARCHAR(10)
);

-- 3. Modifier la table
ALTER TABLE classes ADD salle INT;

-- 4. Ajouter une contrainte
ALTER TABLE classes ADD INDEX idx_nom (nom);

-- 5. Voir la structure
DESCRIBE classes;

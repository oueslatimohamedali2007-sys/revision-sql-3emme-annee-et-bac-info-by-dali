Chapitre 5

Manipulation des données (DML)

Ajouter, modifier ou supprimer des enregistrements de données dans vos tables.

INSERT INTO — Ajouter des Enregistrements

Insère une ou plusieurs nouvelles lignes dans une table.

-- Insérer une ligne avec toutes les colonnes
INSERT INTO eleves VALUES (1, 'Ben Ali', 'Ahmed', '3STI', 15.5);

-- Insérer en spécifiant les colonnes
INSERT INTO eleves (id, nom, prenom, classe, note) 
VALUES (2, 'Trabelsi', 'Fatima', '4STI', 16);

-- Insérer plusieurs lignes à la fois
INSERT INTO eleves (id, nom, prenom, classe, note) VALUES
  (3, 'Dammak', 'Mohamed', '3STI', 14),
  (4, 'Jebari', 'Leila', '4STI', 17),
  (5, 'Samadi', 'Ali', '3STI', 13);

UPDATE — Modifier des Enregistrements

Change les valeurs de colonnes pour les lignes correspondant à une condition.

-- Modifier un élève spécifique
UPDATE eleves SET note = 16 WHERE id = 1;

-- Modifier plusieurs colonnes
UPDATE eleves SET classe = '4STI', note = 18 WHERE nom = 'Ben Ali';

-- Modifier plusieurs lignes avec condition
UPDATE eleves SET note = note + 1 WHERE classe = '3STI';

-- ⚠️ ATTENTION: modifier toute la table sans WHERE
UPDATE eleves SET note = 15; -- Toutes les notes deviennent 15!

DELETE — Supprimer des Enregistrements

Supprime les lignes correspondant à une condition.

-- Supprimer un élève spécifique
DELETE FROM eleves WHERE id = 1;

-- Supprimer par condition
DELETE FROM eleves WHERE classe = '3STI' AND note < 10;

-- Supprimer avec LIKE
DELETE FROM eleves WHERE nom LIKE 'T%';

-- ⚠️ ATTENTION: supprimer sans WHERE supprime TOUTE la table!
DELETE FROM eleves; -- Vide complètement la table!

💡 Bonnes Pratiques

Conseils pour éviter les erreurs lors de modifications:

-- ✓ BON: Toujours vérifier avec SELECT d'abord
SELECT * FROM eleves WHERE classe = '3STI'; -- Vérifier avant suppression
DELETE FROM eleves WHERE classe = '3STI';

-- ✗ MAUVAIS: Pas de condition (danger!)
DELETE FROM eleves; -- Supprime tout!

-- ✓ BON: Utiliser une transaction pour tester
-- BEGIN TRANSACTION; (selon le SGBD)
-- UPDATE eleves SET note = note + 1;
-- ROLLBACK; (annuler si erreur)

-- Insérer avec valeur calculée
INSERT INTO eleves (id, nom, prenom, classe, note)
SELECT id+100, nom, prenom, classe, note FROM eleves_archive;

Cas d'usage: Opération Complète

Exemple combinant insertion, modification et suppression:

-- 1. Ajouter un nouvel élève
INSERT INTO eleves (id, nom, prenom, classe, note) 
VALUES (6, 'Nouvel', 'Élève', '3STI', 12);

-- 2. Corriger une note
UPDATE eleves SET note = 15 WHERE nom = 'Nouvel';

-- 3. Supprimer les mauvais élèves (< 8)
DELETE FROM eleves WHERE note < 8;

-- 4. Vérifier le résultat
SELECT * FROM eleves ORDER BY note DESC;
